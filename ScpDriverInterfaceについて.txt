ScpBus scpBus = new ScpBus();
scpBus.PlugIn(1);
X360Controller controller = new X360Controller();
controller.LeftTrigger = 255;
controller.RightStickX = 32767;
controller.Buttons |= X360Buttons.A;
controller.Buttons ^= X360Buttons.A;
controller.Buttons &= ~X360Buttons.A;
scpBus.Report(1,controller.GetReport());
scpBus.Close();
